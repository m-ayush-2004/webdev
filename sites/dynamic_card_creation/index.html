        void bubble_sort ( int* a, int n ) {
            for  ( int i = 0; i  <  n; i++ ) {
                for  ( int j = 0; j  <  n; j++ ) {
                    if ( a[j] > a[j+1] ) 
                swap ( a,j,j+1 ) ;
                }
            }
            cout<<  "sorted array : "  ;
            for ( int i = 0 ; i < n; i++ ) {
                printf (   "%d "   , * ( a + i )  ) ;
            }
            cout<<  "\n"  ;
        }

        void merge ( int *a,int l,int m, int r ) {
            int size1=m-l+1,size2=r-m;
            int left[size1],right[size2];
            int i,j,o;
            for ( i = 0 ; i < size1; i++ ) {
                left[i]=a[l+i];}
            for ( i = 0 ; i < size2; i++ ) {
                right[i]=a[m+1+i];}
            i=j=0;
            o=l;
            while  ( i  <  size1 && j  <  size2 ) {
                if  ( left[i]    < =   right[j] ) {
                    a[o]=left[i];
                    i++;
                }
                else{
                    a[o]=right[j];
                    j++;
                }
                o++;
            }
            while  ( i  <  size1 )  {
                a[o] = left[i];
                i++;
                o++;
            }
            while  ( j  <  size2 )  {
                a[o] = right[j];
                j++;
                o++;
            }
        }
